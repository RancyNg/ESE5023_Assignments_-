wind['year']= wind['DATE'].map(lambda x:x.split('-')[0])
wind['WND2']=wind['WND1'].astype('int')
lst=['2010','2011','2012','2013','2014','2015','2016','2017','2018','2019','2020']

    table=wind.loc[(wind['year'] == a )].groupby(['month'])['WND2'].mean().reset_index()
    plt.plot(table['month'],table['WND2'])
